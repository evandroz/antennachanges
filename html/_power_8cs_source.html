106  protected HSFProfile<double> CalcSolarPanelPowerProfile(double start, double end, SystemState state, DynamicState position, Universe universe)
107  {
108  // create solar panel profile for this event
109  double freq = 5;
110  ShadowState lastShadow = universe.Sun.castShadowOnPos(position, start);
111  HSFProfile<double> solarPanelPowerProfile = new HSFProfile<double>(start, GetSolarPanelPower(lastShadow));
112 
113  for (double time = start + freq; time <= end; time += freq)
114  {
115  ShadowState shadow = universe.Sun.castShadowOnPos(position, time);
116  // if the shadow state changes during this step, save the power data
117  if (shadow != lastShadow)
118  {
119  solarPanelPowerProfile[time] = GetSolarPanelPower(shadow);
120  lastShadow = shadow;
121  }
122  }
123  state.AddValue(POWIN_KEY, solarPanelPowerProfile);
124  return solarPanelPowerProfile;
125  }

135  public override bool CanPerform(Event proposedEvent, Universe universe)
136  {

139  return false;
140  double es = proposedEvent.GetEventStart(Asset);

142  double ee = proposedEvent.GetEventEnd(Asset);

144 
145  if (ee > SimParameters.SimEndSeconds)

147  Console.WriteLine("Simulation ended");

149  }
150 
151  // get the old DOD
152  double olddod = _newState.GetLastValue(Dkeys.First()).Value;
153 
154  // collect power profile out
155  Delegate DepCollector;
156  SubsystemDependencyFunctions.TryGetValue("DepCollector", out DepCollector);
157  HSFProfile<double> powerOut = (HSFProfile<double>)DepCollector.DynamicInvoke(proposedEvent); // deps->callDoubleDependency("POWERSUB_getPowerProfile");
158  powerOut = powerOut + powerSubPowerOut;
159  // collect power profile in
160  DynamicState position = Asset.AssetDynamicState;
161  HSFProfile<double> powerIn = CalcSolarPanelPowerProfile(es, te, _newState, position, universe);
162  // calculate dod rate
163  HSFProfile<double> dodrateofchange = ((powerOut - powerIn) / _batterySize);
164 
165  bool exceeded= false ;
166  double freq = 1.0;
167  HSFProfile<double> dodProf = dodrateofchange.lowerLimitIntegrateToProf(es, te, freq, 0.0, ref exceeded, 0, olddod);
168 
169  _newState.AddValue(DOD_KEY, dodProf);
170  return true;
171  }

180  public override bool CanExtend(Event proposedEvent, Universe universe, double evalToTime) {

197  // collect power profile in

199  HSFProfile<double> powerIn = CalcSolarPanelPowerProfile(te, ee, proposedEvent.State, position, universe);

204  double freq = 1.0;
205  HSFProfile<double> dodProf = dodrateofchange.limitIntegrateToProf(te, ee, freq, 0.0, 1.0, ref exceeded_lower, ref exceeded_upper, 0, olddod);

207  return false;
