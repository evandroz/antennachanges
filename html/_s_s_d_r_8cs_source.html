68  public override bool CanPerform(Event proposedEvent, Universe environment)

70  if (!base.CanPerform(proposedEvent, environment))

72  if (_task.Type == TaskType.IMAGING)
73  {
74  double ts = proposedEvent.GetTaskStart(Asset);
75  double te = proposedEvent.GetTaskEnd(Asset);
76 
77  double oldbufferratio = _newState.GetLastValue(Dkeys[0]).Value;
78 
79  Delegate DepCollector;

82

84  HSFProfile<double> newdataratio = newdataratein.upperLimitIntegrateToProf(ts, te, 1, 1, ref exceeded, 0, oldbufferratio);
85  if (!exceeded)
86  {
87  _newState.AddValue(DATABUFFERRATIO_KEY, newdataratio);
88  return true;
89  }
90 
91  // TODO: Change to logger
92  Console.WriteLine("SSDR buffer full");
93  return false;
94  }
95  else if (_task.Type == TaskType.COMM)
96  {
97  double ts = proposedEvent.GetTaskStart(Asset);
98  proposedEvent.SetTaskEnd(Asset, ts + 60.0);
99  double te = proposedEvent.GetTaskEnd(Asset);
100 
101  double data = _bufferSize * _newState.GetLastValue(Dkeys.First()).Value;
102  double dataqueout = data / 2 > 50 ? data / 2 : data;
103 
104  if (data - dataqueout < 0)
105  dataqueout = data;
106 
107  if (dataqueout > 0)
108  _newState.AddValue(DATABUFFERRATIO_KEY, new KeyValuePair<double, double>(te, (data - dataqueout) / _bufferSize));
109  return true;
110  }
111  return true;
