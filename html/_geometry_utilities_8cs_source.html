15  public static bool hasLOS(Matrix<double> posECI1, Matrix<double> posECI2)
16  {
17  // calculate the minimum distance to the center of the earth
18  double d = Matrix<double>.Norm(Matrix<double>.Cross(posECI2 - posECI1, posECI1)) / Matrix<double>.Norm(posECI2 - posECI1);
19  /* parameter t is is the parameter that represents where the minimum distance is
20  d is a minimum at (posECI1) + t*(posECI2 - posECI1)*/
21  double t = -Matrix<double>.Dot(posECI1, posECI2 - posECI1) / System.Math.Pow(Matrix<double>.Norm(posECI2 - posECI1), 2);
22  /* if t > 1 or t < 0, then the minumim distance does not occur along the line segment connecting positions 1 and 2,
23  and the two positions are visible from each other*/
24  if (t >= 1 || t <= 0)
25  return true;
26  // otherwise compare the minimum distance to the radius of the earth
27  return d >= SimParameters.EARTH_RADIUS;
28  }
