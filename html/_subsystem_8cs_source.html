110  public virtual HSFProfile<double> DependencyCollector(Event currentEvent)
111  {
112  if (SubsystemDependencyFunctions.Count == 0)
113  throw new MissingMemberException("You may not call the dependency collector in your can perform because you have not specified any dependency functions for " + Name);
114  HSFProfile<double> outProf = new HSFProfile<double>();
115  foreach (var dep in SubsystemDependencyFunctions)
116  {
117  if (!dep.Key.Equals("DepCollector"))
118  {
119  HSFProfile<double> temp = (HSFProfile<double>)dep.Value.DynamicInvoke(currentEvent);
120  outProf = outProf + temp;
121  }
122  }
123  return outProf;
124  }
