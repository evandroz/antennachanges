106  public override bool CanPerform(Event proposedEvent, Universe environment)
107  {
108  if (!base.CanPerform(proposedEvent, environment))
109  return false;
110  if (_task.Type == TaskType.IMAGING)
111  {
112  //set pixels and time to caputre based on target value
113  int value = _task.Target.Value;
114  double pixels = _lowQualityPixels;
115  double timetocapture = _lowQualityTime;
116  if (value <= _highQualityTime && value >= _midQualityTime) //Morgan took out magic numbers
117  {
118  pixels = _midQualityPixels;
119  timetocapture = _midQualityTime;
120  }
121  if (value > _highQualityTime)
122  {
123  pixels = _highQualityPixels;
124  timetocapture = _highQualityTime;
125  }
126 
127  // get event start and task start times
128  double es = proposedEvent.GetEventStart(Asset);
129  double ts = proposedEvent.GetTaskStart(Asset);
130  double te = proposedEvent.GetTaskEnd(Asset);
131  if (ts > te)
132  {
133  // TODO: Change this to Logger
134  Console.WriteLine("EOSensor lost access");
135  return false;
136  }
137 
138  // set task end based upon time to capture
139  te = ts + timetocapture;
140  proposedEvent.SetTaskEnd(Asset, te);
141 
142  // calculate incidence angle
143  // from Brown, Pp. 99
144  DynamicState position = Asset.AssetDynamicState;
145  double timage = ts + timetocapture / 2;
146  Matrix<double> m_SC_pos_at_tf_ECI = position.PositionECI(timage);
147  Matrix<double> m_target_pos_at_tf_ECI = _task.Target.DynamicState.PositionECI(timage);
148  Matrix<double> m_pv = m_target_pos_at_tf_ECI - m_SC_pos_at_tf_ECI;
149  Matrix<double> pos_norm = -m_SC_pos_at_tf_ECI / Matrix<double>.Norm(-m_SC_pos_at_tf_ECI);
150  Matrix<double> pv_norm = m_pv / Matrix<double>.Norm(m_pv);
151 
152  double incidenceang = 90 - 180 / Math.PI * Math.Acos(Matrix<double>.Dot(pos_norm, pv_norm));
153 
154  // set state data
155  _newState.AddValue(INCIDENCE_KEY, new KeyValuePair<double, double>(timage, incidenceang));
156  _newState.AddValue(INCIDENCE_KEY, new KeyValuePair<double, double>(timage + 1, 0.0));
157 
158  _newState.AddValue(PIXELS_KEY, new KeyValuePair<double, double>(timage, pixels));
159  _newState.AddValue(PIXELS_KEY, new KeyValuePair<double, double>(timage + 1, 0.0));
160 
161  _newState.AddValue(EOON_KEY, new KeyValuePair<double, bool>(ts, true));
162  _newState.AddValue(EOON_KEY, new KeyValuePair<double, bool>(te, false));
163  }
164  return true;
165 
166  }
