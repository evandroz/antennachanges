186  public void CropSchedules(List<SystemSchedule> schedulesToCrop, Evaluator scheduleEvaluator, SystemSchedule emptySched)
187  {

189  foreach (SystemSchedule systemSchedule in schedulesToCrop)

192  // Sort the sysScheds by their values
