576  public static SystemState setInitialSystemState(List<XmlNode> ICNodes, Asset asset)
577  {
578  // Set up Subsystem Nodes, first loop through the assets in the XML model input file
579  //int n = modelInputXMLNode.ChildNode("ASSET");

589  StateVarKey<int> svk = new StateVarKey<int>(key);
590  state.AddValue(svk, new KeyValuePair<double, int>(SimParameters.SimStartSeconds, val));
591  }

600  {
601  double val;
602  Double.TryParse(ICNode.Attributes["value"].Value, out val);
603  StateVarKey<double> svk = new StateVarKey<double>(key);
604  state.AddValue(svk, new KeyValuePair<double, double>(SimParameters.SimStartSeconds, val));
605  }
606  else if (type.Equals("Bool"))
607  {
608  string val = ICNode.Attributes["value"].Value;
609  bool val_ = false;
610  if (val.Equals("True") || val.Equals("1"))
611  val_ = true;
612  StateVarKey<bool> svk = new StateVarKey<bool>(key);
613  state.AddValue(svk, new KeyValuePair<double, bool>(SimParameters.SimStartSeconds, val_));
614  }
615  else if (type.Equals("Matrix"))
616  {
617  Matrix<double> val = new Matrix<double>(ICNode.Attributes["value"].Value);
618  StateVarKey<Matrix<double>> svk = new StateVarKey<Matrix<double>>(key);
619  state.AddValue(svk, new KeyValuePair<double, Matrix<double>>(SimParameters.SimStartSeconds, val));
620  }
621  else if (type.Equals("Quat"))
622  {
623  // Quaternions still need an initializer from a string, like Matrices
624  }
625  }
626  return state;
627  }
