44  public SystemSchedule(StateHistory oldStates, Stack<Access> newAccessList, double newEventStartTime)
45  {
46  Dictionary<Asset, Task> tasks = new Dictionary<Asset, Task>();
47  Dictionary<Asset, double> taskStarts = new Dictionary<Asset, double>();
48  Dictionary<Asset, double> taskEnds = new Dictionary<Asset, double>();
49  Dictionary<Asset, double> eventStarts = new Dictionary<Asset, double>();
50  Dictionary<Asset, double> eventEnds = new Dictionary<Asset, double>();
51 
52  foreach (var access in newAccessList)
53  {
54  if (access.Task != null)
55  {
56  if (access.AccessStart <= newEventStartTime && newEventStartTime <= access.AccessEnd)
57  taskStarts.Add(access.Asset, newEventStartTime);
58  else if (access.AccessStart >= newEventStartTime && access.AccessStart <= newEventStartTime + SchedParameters.SimStepSeconds)
59  taskStarts.Add(access.Asset, access.AccessStart);
60  else
61  {
62  Console.WriteLine("Event Start: " + newEventStartTime + " AccesStart: " + access.AccessStart + " AccessEnd: " + access.AccessEnd);
63  taskStarts.Add(access.Asset, newEventStartTime);
64  }
65  tasks.Add(access.Asset, access.Task);
66  if (access.AccessEnd > SimParameters.SimEndSeconds)
67  taskEnds.Add(access.Asset, SimParameters.SimEndSeconds);
68  else
69  taskEnds.Add(access.Asset, access.AccessEnd);
70  eventStarts.Add(access.Asset, newEventStartTime);
71  if (newEventStartTime + SchedParameters.SimStepSeconds > SimParameters.SimEndSeconds)
72  eventEnds.Add(access.Asset, SchedParameters.SimStepSeconds);
73  else
74  eventEnds.Add(access.Asset, newEventStartTime + SchedParameters.SimStepSeconds);
75  }
76  else
77  {
78  taskStarts.Add(access.Asset, newEventStartTime);
79  taskEnds.Add(access.Asset, newEventStartTime);
80  tasks.Add(access.Asset, null);
81  eventStarts.Add(access.Asset, newEventStartTime);
82  eventEnds.Add(access.Asset, newEventStartTime + SchedParameters.SimStepSeconds);
83  }
84 
85  }
86  Event eventToAdd = new Event(tasks, new SystemState(oldStates.GetLastState())); //all references
87  eventToAdd.SetEventEnd(eventEnds);
88  eventToAdd.SetTaskEnd(taskEnds);
89  eventToAdd.SetEventStart(eventStarts);
90  eventToAdd.SetTaskStart(taskStarts);
91  AllStates = new StateHistory(oldStates, eventToAdd);
92  }

101  public bool CanAddTasks(Stack<Access> newAccessList, double newTaskStartTime)
