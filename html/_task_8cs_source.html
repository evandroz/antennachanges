49  public static bool loadTargetsIntoTaskList(XmlNode targetDeckXMLNode, Stack<Task> tasks)
50  {
51  if (targetDeckXMLNode == null)
52  return false;
53  log.Info("Loading target deck...");
54  int maxTimesPerform = 1;
55  bool allLoaded = true;
56  string targetType, taskType;
57  foreach (XmlNode targetNode in targetDeckXMLNode.ChildNodes)
58  {
59  if (targetNode.Attributes["TaskType"] != null)
60  taskType = targetNode.Attributes["TaskType"].Value.ToString();
61  else {
62  log.Fatal("Missing Task Type");
63  return false;
64  }
65  var taskTypeEnum = (TaskType)Enum.Parse(typeof(TaskType), taskType);
66  if (targetNode.Attributes["MaxTimes"] != null)
67  {
68  Int32.TryParse(targetNode.Attributes["MaxTimes"].Value.ToString(), out maxTimesPerform);
69  tasks.Push(new Task(taskTypeEnum, new Target(targetNode), maxTimesPerform));
70  }
71  else
72  tasks.Push(new Task(taskTypeEnum, new Target(targetNode), maxTimesPerform));
73  }
74  log.Info("Number of Targets Loaded: "+ tasks.Count);
75 
76  return allLoaded;
77  }
