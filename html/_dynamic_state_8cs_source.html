173  private void PropagateState(double simTime)
174  {
175  log.Info("Integrating and resampling dynamic state data to "+ simTime + "seconds...");
176  Matrix <double> tSpan = new Matrix<double>(new double[1, 2] { { _stateData.Last().Key, simTime } });
177  // Update the integrator parameters using the information in the XML Node
178 
179  Matrix<double> data = Integrator.RK45(Eoms, tSpan, InitialConditions(), _integratorOptions);
180 
181  for (int index = 1; index <= data.Length; index++)
182  _stateData[data[1, index]] = data[new MatrixIndex(2, data.NumRows), index];
183 
184  log.Info("Done Integrating");
185  }

206 
207  else if (Type == DynamicStateType.STATIC_ECI)

209  else if (Type == DynamicStateType.PREDETERMINED_ECI || Type == DynamicStateType.PREDETERMINED_LLA)

212  if (hasNotPropagated)
213  PropagateState(SimParameters.SimEndSeconds);
214 
215  Matrix<double> dynamicStateAtSimTime;
216 
217  if (!_stateData.TryGetValue(simTime, out dynamicStateAtSimTime))
218  {
219  int lowerIndex = _stateData.Keys.LowerBoundIndex(simTime);
220  int slopeInd = 1;
221  if (simTime >= SimParameters.SimEndSeconds)
222  slopeInd = -1;
223  KeyValuePair<double, Matrix<double>> lowerData = _stateData.ElementAt(lowerIndex);
224  KeyValuePair<double, Matrix<double>> upperData = _stateData.ElementAt(lowerIndex + slopeInd);
225 
226  double lowerTime = lowerData.Key;
227  Matrix<double> lowerState = lowerData.Value;
228 
229  double upperTime = upperData.Key;
230  Matrix<double> upperState = upperData.Value;
231 
232  Matrix<double> slope = (upperState - lowerState) / (upperTime - lowerTime);
233 
234  dynamicStateAtSimTime = slope * (simTime - lowerTime) + lowerState;
235  }
236

239  }
